### LangChain + LM Studio 패턴

```python
from langchain_openai import ChatOpenAI
from langchain.prompts import ChatPromptTemplate
from langchain.output_parsers import JsonOutputParser
from pydantic import BaseModel
import os

class FinancialAdvice(BaseModel):
    summary: str
    risk_level: str
    recommendations: list[str]

llm = ChatOpenAI(
    base_url=os.getenv("LM_STUDIO_URL", "http://localhost:1234/v1"),
    api_key="lm-studio",
    model=os.getenv("LM_STUDIO_MODEL", "local-model"),
    temperature=0.3,
)

parser = JsonOutputParser(pydantic_object=FinancialAdvice)
prompt = ChatPromptTemplate.from_messages([
    ("system", "금융 상담 전문가로서 JSON 형식으로 답변하세요.\n{format_instructions}"),
    ("user", "{question}"),
]).partial(format_instructions=parser.get_format_instructions())

chain = prompt | llm | parser
result = chain.invoke({"question": "30대 직장인에게 맞는 투자 포트폴리오는?"})
```

### RAG + LM Studio 전체 흐름

```python
from sentence_transformers import SentenceTransformer
import chromadb
from lmstudio_config import call_lm

# 1. 임베딩 모델 (로컬, 무료)
embedder = SentenceTransformer("paraphrase-multilingual-MiniLM-L12-v2")

# 2. 벡터DB 초기화
db = chromadb.Client()
collection = db.get_or_create_collection("docs")

# 3. 문서 인덱싱
documents = ["RAG 는 검색 증강 생성입니다.", "LM Studio 는 로컬 LLM 서버입니다."]
embeddings = embedder.encode(documents).tolist()
collection.add(documents=documents, embeddings=embeddings, ids=["d1", "d2"])

# 4. 쿼리 및 검색
query = "RAG 가 무엇인가요?"
q_emb = embedder.encode([query]).tolist()
results = collection.query(query_embeddings=q_emb, n_results=2)
context = "\n".join(results["documents"][0])

# 5. LM Studio 생성
answer = call_lm([
    {"role": "system", "content": "컨텍스트만 사용해서 답변하세요."},
    {"role": "user", "content": f"컨텍스트:\n{context}\n\n질문: {query}"},
], temperature=0.2, max_tokens=300)
print(answer)
```
